V2.7 完成了核心终局内容和挑战系统。V2.8 将聚焦于：

4. **战术深度深化**：指挥官系统、阵型系统、环境交互
5. **平台适配准备**：Steam 成就、云存档、手柄支持

function SeasonTaskSystem.checkTaskProgress(taskId, currentValue)
    local task = SEASON_TASKS[taskId]
    if not task then return end

    local progress = SEASON_STATE.tasksCompleted[taskId] or 0
    progress = math.min(progress + currentValue, task.target)

    SEASON_STATE.tasksCompleted[taskId] = progress

    if progress >= task.target and not SEASON_STATE.tasksClaimed[taskId] then
        -- 任务完成但未领取，标记为可领取
        SeasonTaskSystem.markTaskComplete(taskId)
    end
end

**指挥官技能效果**：
```lua
function CommanderSystem.activateSkill(commanderId, skillId, ctx)
    local commander = COMMANDERS[commanderId]
    if not commander then return end

    local skill = nil
    for _, s in ipairs(commander.skills) do
        if s.id == skillId then skill = s; break end
    end
    if not skill then return end

    -- 技能冷却
    if commander.skillCooldowns and commander.skillCooldowns[skillId] > 0 then
        return false, "技能冷却中"
    end

    -- 应用效果
    if skill.effect.dmgReduction then
        -- 战术撤退：降低伤害
        local buff = {
            type = "DMG_REDUCTION",
            value = skill.effect.dmgReduction,
            duration = skill.effect.duration,
            name = skill.name,
        }
        PlayerFleet:applyBuff(buff)
    elseif skill.effect.atkBoost then
        -- 鼓舞士气：提升攻击
        local buff = {
            type = "ATK_BOOST",
            value = skill.effect.atkBoost,
            duration = skill.effect.duration,
            name = skill.name,
        }
        PlayerFleet:applyBuff(buff)
    elseif skill.effect.dmgMult then
        -- 毁灭打击：对敌舰造成高伤害
        local target = BattleAI.getHighestThreatEnemy()
        if target then
            BattleSystem.dealDamage(target, target.health * (skill.effect.dmgMult - 1))
        end
    end

    -- 设置冷却
    commander.skillCooldowns = commander.skillCooldowns or {}
    commander.skillCooldowns[skillId] = skill.cooldown

    return true, "技能激活"
end

-- 被动技能效果
function CommanderSystem.applyPassiveEffects(commanderId, ctx)
    local commander = COMMANDERS[commanderId]
    if not commander or not commander.passive then return end

    local passive = commander.passive

    if passive.effects.buildSpeedMult then
        ctx.playerState.buildSpeedMult = (ctx.playerState.buildSpeedMult or 1.0)
            * passive.effects.buildSpeedMult
    end

    if passive.effects.lowHpDmgBoost then
        ctx.playerState.lowHpDmgBoost = passive.effects.lowHpDmgBoost
    end
end
```

function GuildSystem.createGuild(name, tag, playerState)
    if playerState.guildId then
        return false, "已在公会中"
    end

    local guildId = "GUILD_" .. UUID()
    GUILDS[guildId] = {
        id = guildId,
        name = name,
        tag = tag,
        leaderId = playerState.id,
        members = {
            { id = playerState.id, name = playerState.name,
              role = "LEADER", joinTime = os.time(), contribution = 0 }
        },
        level = 1,
        exp = 0,
        maxMembers = 10,
        createdAt = os.time(),
    }

    playerState.guildId = guildId
    return true, guildId
end

function GuildSystem.joinGuild(guildId, playerState)
    local guild = GUILDS[guildId]
    if not guild then return false, "公会不存在" end

    if playerState.guildId then
        return false, "已在公会中"
    end

    if #guild.members >= guild.maxMembers then
        return false, "公会已满"
    end

    if guild.settings.joinType == "APPROVAL" then
        -- 需要审批，加入申请列表
        guild.applications = guild.applications or {}
        table.insert(guild.applications, {
            playerId = playerState.id,
            playerName = playerState.name,
            applyTime = os.time(),
            message = "申请加入公会",
        })
        return false, "已提交申请，等待审批"
    elseif guild.settings.joinType == "OPEN" then
        table.insert(guild.members, {
            id = playerState.id,
            name = playerState.name,
            role = "MEMBER",
            joinTime = os.time(),
            contribution = 0,
        })
        playerState.guildId = guildId
        return true, "加入成功"
    end

    return false, "无法加入"
end

function GuildSystem.contribute(guildId, playerId, resourceType, amount)
    local guild = GUILDS[guildId]
    if not guild then return false, "公会不存在" end

    local member = nil
    for _, m in ipairs(guild.members) do
        if m.id == playerId then member = m; break end
    end
    if not member then return false, "不是公会成员" end

    -- 消耗资源
    if not ResourceManager.spend(playerId, resourceType, amount) then
        return false, "资源不足"
    end

    -- 增加贡献
    member.contribution = member.contribution + amount
    guild.exp = guild.exp + amount

    -- 检查升级
    GuildSystem.checkLevelUp(guildId)

    return true, "贡献成功"
end

function GalaxyEventSystem.triggerEvent(eventId)
    local event = GALAXY_EVENTS[eventId]
    if not event then return false, "事件不存在" end

    -- 应用事件效果
    for _, effect in ipairs(event.effects) do
        GalaxyEventSystem.applyEffect(effect)
    end

    -- 记录活跃事件
    local activeEvent = {
        id = eventId,
        startTime = os.time(),
        endTime = event.duration > 0 and (os.time() + event.duration) or nil,
        effects = event.effects,
    }
    table.insert(GALAXY_STATE.activeEvents, activeEvent)

    -- 触发事件通知
    NotificationSystem.push({
        type = "EVENT",
        title = event.name,
        message = event.desc,
        icon = event.icon,
        duration = 5,
    })

    return true, "事件触发"
end

function GalaxyEventSystem.randomEventCheck()
    -- 每 30 秒检查是否触发事件
    for eventId, event in pairs(GALAXY_EVENTS) do
        if math.random() < event.probability then
            GalaxyEventSystem.triggerEvent(eventId)
            break
        end
    end
end

## 9. 技术风险与缓解

| 风险 | 等级 | 说明 | 缓解措施 |
|------|------|------|---------|
| 赛季积分通胀 | 高 | 玩家刷任务导致积分过多 | 任务每日/每周重置上限，会员赛季限定 |
| 公会数据冲突 | 中 | 多玩家同时操作导致数据不一致 | 使用操作日志和锁机制 |
| 剧情分支存储膨胀 | 中 | 多分支导致存档过大 | 仅存储关键分支点，剧情静态生成 |
| Steam SDK 集成失败 | 中 | Steam API 调用失败 | 提供离线模式作为后备 |
| 手柄适配不完整 | 低 | 部分功能无法用手柄操作 | 提供完整的键鼠+手柄双支持 |
| 银河事件影响失衡 | 中 | 某些事件组合导致游戏过易 | 事件互斥列表，最大同时生效数限制 |
